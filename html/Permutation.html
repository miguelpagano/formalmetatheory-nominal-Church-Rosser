lemmaπ⁻¹∘π≡id : {π : Π}{M : Λ} →  (π ⁻¹ ∙ π ∙ M) ≡ M

lemmaπ⁻¹∘π≡id {(a , b) ∷ π}  {M}
  = begin≡
      ((a , b) ∷ π) ⁻¹ ∙ ((a , b) ∷ π) ∙ M
    ≡⟨ refl ⟩
      ((a , b) ∷ π) ⁻¹ ∙ （ a ∙ b ） (π ∙ M)
    ≡⟨ cong (λ t → t ∙ （ a ∙ b ） (π ∙ M)) (unfold-reverse (a , b) π)  ⟩

    ≡⟨  lemmaπ∙π′∙M≡π++π′∙M {π ⁻¹ ++ [ (a , b) ]} ⟩

      π ⁻¹ ++ ( (a , b) ∷ (a , b) ∷ []) ∙ (π ∙ M)
